** 快速排序
   #+BEGIN_SRC emacs-lisp
     (defun quicksort (lst)
       "Implement the quicksort algorithm."
       (if (null lst) nil
         (let* ((spl (car lst))
                (rst (cdr lst))
                (smalp (lambda (x)
                         (< x spl))))
           (append (quicksort (remove-if-not smalp rst))
                   (list spl)
                   (quicksort (remove-if smalp rst))))))

     (quicksort '(5 7 1 3 -9 8 7 -4 0))
     ==> (-9 -4 0 1 3 5 7 7 8)
   #+END_SRC
